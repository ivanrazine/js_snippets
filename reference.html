JavaScript Snippets

    Loop

    jQuery

    
        $(el).each(function() {
            ...
        });

    Vanilla

    
    var blocks = document.querySelectorAll(el);

    [].forEach.call(blocks, function (el) {
        ...
    });

    Style

    jQuery

    
    $(el).css("background", "red");

    Vanilla

    
    el.style.backgroundColor = "red";

    Class

    jQuery

    
    $(el).addClass("blue");
    $(el).removeClass("block");
    $(el).toggleClass("toggled");

    Vanilla

    
    el.classList.add("blue");
    el.classList.remove("block");
    el.classList.toggle("toggled");

    Attribute

    jQuery

    
    $(el).attr("title","My fancy block");
    $(el).data("title","My fancy block");
    var a = $(el).data("title");

    Vanilla

    
    el.setAttribute("title","My fancy block");
    el.dataset.title = "Another title";
    var a = el.getAttribute("title");

    Inner text & HTML

    jQuery

    
    $(el).text("Hello World");
    $(el).html("Hello World");

    Vanilla

    
    el.textContent = "Hello World";
    el.innerHTML = "Hello World";

    Append & prepend

    jQuery

    
    $(el).append("*-- ");
    $(el).prepend(" --*");

    Vanilla

    
    el.innerHTML += " --*";

    var oldHTML = el.innerHTML;
    el.innerHTML = "*-- " + oldHTML;

    First & last

    jQuery

    
    $(el).first();
    $(el).last();

    Vanilla

    
    var lastDiv = document.querySelectorAll(el);
    lastDiv = lastDiv[lastDiv.length- 1];

    var firstDiv = document.querySelectorAll(el)[0];
